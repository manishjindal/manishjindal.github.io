#include<iostream>
using namespace std;
int main()
{
      int n;
      cin>>n;
      char a[1000000];    
      cin>>a;
      
      n=n*2;
      
      char b,c;
      b='X';
      for(int i=0;i<n;i=i+2)
      {    
          
           if(a[i]!='A' && b!='A' && a[i+1]!='A')
               { cout<<"A"; c='A';}
           else  if(a[i]!='B' && b!='B' && a[i+1]!='B')
                {cout<<"B"; c='B';}
           else  if(a[i]!='C' && b!='C' && a[i+1]!='C')
                {cout<<"C"; c='C';}
           else  if(a[i]!='D' && b!='D' && a[i+1]!='D')
                {cout<<"D"; c='D';}
                
               // cout<<"c="<<c<<endl;cout<<"a[i]="<<a[i]<<endl;cout<<"a[i+1]="<<a[i+1]<<endl;
            if(a[i]!='A' && c!='A' && a[i+1]!='A')
               { cout<<"A"; b='A';}
           else  if(a[i]!='B' && c!='B' && a[i+1]!='B')
                {cout<<"B"; b='B';}
           else  if(a[i]!='C' && c!='C' && a[i+1]!='C')
                {cout<<"C"; b='C';}
           else  if(a[i]!='D' && c!='D' && a[i+1]!='D')
                {cout<<"D"; b='D';}
                
    
    
      }
    
    
    return 0;
}
