sub concatenate {
	my $a = "abc";
	my $b = "def";

	$c = $a . $b;

	print $c;

}

sub repeatation {

	my $a = 'a';

	print "double is =" . $a x 2 . "\n";
	print "triple is = " . $a x 3 . "\n";
}

sub printRange {
	print (2..10);
}
