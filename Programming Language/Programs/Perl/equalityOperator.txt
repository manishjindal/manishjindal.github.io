sub checkNumericEqual {
	my $a = 5;
	my $b = $a;

	if ($a == $b) {
		print "Equal";
	}else {
		print "Not Equal";
	}

	if ($a eq $b) {
		print "Equal";
	}else {
		print "Not Equal";
	}
}

sub checkStringEqual {
	my $a = "abc";
	my $b = $a;

	if ($a eq $b) {
		print "Equal";
	}else {
		print "Not Equal";
	}

}

checkNumericEqual();
checkStringEqual();
